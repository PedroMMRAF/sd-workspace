# Maven compile
mvn clean compile assembly:single docker:build

# Create network
docker network create -d bridge sdnet

# Run server
docker run --rm -h <HOST-NAME> --name <NAME> --network sdnet -p 8080:8080 <PROJECT> <COMANDO>

# Run client
docker run --rm -it --name <NAME> --network sdnet <PROJECT> /bin/bash

# Run client command
java -cp ./sd2223.jar "<COMMAND-NAME>" <ARGS>
